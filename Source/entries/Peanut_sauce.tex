\begin{recipe}[PeanutSauce]{Peanut Sauce}{Half a cup}{10 minutes}
	% From september 2017 delicious living magazine

\Ingredient{\fr13 cup plain peanut butter}
\Ingredient{1 tablespoon minced garlic}
\Ingredient{1 tablespoon soy sauce}
\Ingredient{1 tablespoon peanut oil}
\Ingredient{1 tablespoon sesame oil}
\Ingredient{1 tablespoon rice wine vinegar}
\Ingredient{2 teaspoon honey}
\Ingredient{\fr34 teaspoon ground ginger}
Combine in a medium bowl. whisk to combine.

\Ingredient{5 tablespoons water}
\Ingredient{\fr14 teaspoon Sriracha}
Add and whisk again to mix.
