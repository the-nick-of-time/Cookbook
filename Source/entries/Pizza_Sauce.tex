\begin{recipe}[PizzaSauce]{\textbf{Pizza Sauce}}{2 full-size pizzas}{10 minutes}
	\Ingredient{1 can tomato paste}
	\Ingredient{2 teaspoons oregano}
	\Ingredient{1 teaspoon thyme}
	\Ingredient{\fr12 teaspoon basil}
	\Ingredient{1 tablespoon water (optional)}
	Mix together. The water is optional, but makes it easier to get a consistent spread.
\end{recipe}
