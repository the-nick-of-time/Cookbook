\begin{recipe}{Pumpkin Bread}{1 loaf}{20 minutes prep + 60 minutes cooking}
	Have all ingredients at room temperature. Position a rack in the lower third of the oven. Preheat the oven to 350\0F. Grease a 9\X5-inch (8-cup) loaf pan.

	\Ingredient{1 cup \ap}
	\Ingredient{\fr12 cup \ww}
	\Ingredient{1 teaspoon baking soda}
	\Ingredient{\fr14 teaspoon baking powder}
	\Ingredient{\fr12 teaspoon salt}
	\Ingredient{\fr32 teaspoon cinnamon}
	\Ingredient{1 teaspoon ginger}
	\Ingredient{\fr12 teaspoon nutmeg}
	\Ingredient{\fr14 teaspoon cloves}
	Whisk dry ingredients together.

	\Ingredient{\fr13 cup milk}
	\Ingredient{\fr12 teaspoon vanilla}
	Combine liquids in another bowl.

	\Ingredient{6 tablespoons butter}
	In a large bowl, beat the butter until creamy. It should be smooth, and have a little air trapped inside.

	\Ingredient{1 cup white sugar}
	\Ingredient{\fr13 cup brown sugar}
	Beat sugar into the butter.

	\Ingredient{2 eggs}
	Beat into the sugar/butter mixture.

	\Ingredient{1 cup pumpkin pur\'ee}
	Also beat in pumpkin.

	\newstep
	Stir in the flour mixture in three parts, alternating with the milk mixture.
	% The batter should be made homogenous between each addition.

	\Ingredient{\fr12 cup chopped pecans or walnuts}
	\Ingredient{\fr13 cup dried cranberries}
	Add in extra ingredients if desired.

	\newstep
	Pour the batter into the pan and spread it evenly. The batter should be thin enough that it spreads on its own, but a little help can be good. If splitting the recipe between multiple pans/muffin tins, each should be \fr12-\fr34 full.
	Bake \tooth{40 to 60}. Let cool for 5 to 10 minutes before removing from the pan.
\end{recipe}
