\begin{recipe}{Pad Thai}{6 servings}{40 minutes}

\Ingredient{4 ounces thick rice noodles}
Submerge noodles in a large bowl full of hot water. Let rest until soft and white, about 20 minutes.

\Ingredient{\fr92 tablespoons lime juice (about 2 limes)}
\Ingredient{\fr92 tablespoons tamari}
\Ingredient{\fr92 tablespoons dark brown sugar}
\Ingredient{3 tablespoons water}
\Ingredient{5 large cloves of garlic, minced}
In a medium bowl, combine and whisk until the sugar dissolves.

\Ingredient{1 pound string beans}
\Ingredient{1 bunch (about 6) scallions, with ends trimmed}
\Ingredient{2 tablespoons sesame oil}
Cut into 1-inch pieces.
Heat a large wok or skillet. Add the oil and stir fry the vegetables for 5 minutes.

% \Ingredient{1 pound tofu}
\Ingredient{1 pound chicken or tofu, cubed}
\Ingredient{1 egg, beaten}
Add and stir fry for 3 minutes.

\Ingredient{12 ounces mung bean sprouts}
Drain noodles and add these and the sauce to pan. Cook, tossing consistently, until the whole mixture is heated through, about 5 minutes.

\Ingredient{1 cup chopped fresh cilantro}
\Ingredient{\fr12 cup dry-roasted peanuts}
\Ingredient{1 fresh lime, cut into 6 slices (for garnish)}
Add the peanuts and cilantro and serve with lime wedges.

\end{recipe}
