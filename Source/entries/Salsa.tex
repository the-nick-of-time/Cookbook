\begin{recipe}{\textbf{Serrano Salsa}}{2 cups}{30 minutes}
	\Ingredient{12 ounces tomatoes}
	\Ingredient{3-5 fresh serrano chiles, stemmed}
	\Ingredient{\fr13 bunch cilantro}
	\Ingredient{3 garlic cloves}
	Finely dice and mix together. Or for the lazy, use a food processor.

	\Ingredient{1 small (4 ounces) white onion}
	Finely dice and mix in; a food processor will break too much of the cells and make it taste weird.

	\Ingredient{\fr32 teaspoons fresh lime juice}
	\Ingredient{\fr34 teaspoon salt}
	Season to taste.

\end{recipe}
