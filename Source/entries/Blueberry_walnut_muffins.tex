\begin{recipe}{Blueberry-Walnut Wheat Germ Muffins}{16 muffins}{1 hour}
% From January 2011 something magazine

Preheat oven to 400\0F.

\Ingredient{1 cup \ap}
\Ingredient{\fr12 cup \ww}
\Ingredient{1 cup wheat germ}
\Ingredient{2 teaspoons baking powder}
\Ingredient{1 teaspoon salt}
Combine in a large bowl.

\Ingredient{\fr12 cup finely chopped walnuts}
Stir in the walnuts.

\Ingredient{1 egg, well beaten}
\Ingredient{\fr14 cup canola oil}
\Ingredient{\fr14 cup honey}
\Ingredient{\fr14 cup molasses}
In a medium bowl, beat together.

\Ingredient{1 cup milk}
\Ingredient{1 teaspoon vanilla extract}
Add to the wet ingredients.

Gradually add to the dry ingredients and stir until just mixed.

\Ingredient{1 cup fresh or frozen blueberries}
Gently fold in blueberries.

Line muffin tins and divide batter between cups. Bake \tooth{18-20}. Remove and let cool for a few minutes before taking them out.


\end{recipe}
