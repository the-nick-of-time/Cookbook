\begin{recipe}{Banana Bread}{1 loaf}{20 minutes prep, 55 minutes baking}
	Grease a 6-cup loaf pan and preheat oven to 350\0F.

	\Ingredient{\fr43 cups \ap}
	\Ingredient{\fr34 teaspoon salt}
	\Ingredient{\fr12 teaspoon baking soda}
	\Ingredient{\fr14 teaspoon baking powder}
	\Ingredient{\fr14 teaspoon ginger}
	\Ingredient{\fr14 teaspoon nutmeg}
	\Ingredient{\fr18 teaspoon cinnamon}
	Whisk together.

	\Ingredient{\fr{11}{2} unsalted butter}
	\Ingredient{\fr23 cup sugar}
	Beat together. Beat in the flour mixture by parts.

	\Ingredient{2 eggs}
	Gradually beat in.

	\Ingredient{1 cup (about 2) mashed bananas}
	\Ingredient{\fr12 cup chopped walnuts or pecans}
	Fold in until combined. You will end up with a thick batter.
\end{recipe}
