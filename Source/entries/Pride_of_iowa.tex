\begin{recipe}{Pride of Iowa Cookies}{24 cookies}{25 minutes prep, 15 minutes baking}

	\Ingredient{1 cup brown sugar}
	\Ingredient{1 cup white sugar}
	\Ingredient{1 cup shortening (can be butter)}
	Cream together shortening and sugars.

	\Ingredient{2 eggs}
	\Ingredient{1 teaspoon vanilla}
	Beat in eggs and vanilla.

	\Ingredient{1 teaspoon soda}
	\Ingredient{1 teaspoon baking powder}
	\Ingredient{1 teaspoon salt}
	\Ingredient{2 cup flour}
	Mix together dry ingredients, and add them to the wet ingredients. Mix together thoroughly.

	\Ingredient{1 cup coconut (I use 2) (optional, not included in refrigerator variant)}
	\Ingredient{\fr12 cup chopped nuts (black walnuts are best, and I use a generous half cup)}
	\Ingredient{2 cup quick-cooking oatmeal (optional, not included in refrigerator variant)}
	Add the nuts, oatmeal, and coconut. Form the dough into rolls, wrap in plastic, and chill until the dough is firm.\\
	When ready, slice the dough into cookies and bake at 375\0F. When done, they are golden brown and crispy.

\end{recipe}
