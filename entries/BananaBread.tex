\begin{recipe}{Banana Bread}{1 loaf}{20 minutes prep, 55 minutes baking}
	% a blend of Joy of Cooking and https://www.allrecipes.com/recipe/20144/banana-banana-bread/
	Grease a 9\X5 inch loaf pan and preheat oven to 350\0F.

	\Ingredient{\flour{all purpose}{2}}
	\Ingredient{\fr14 teaspoon salt}
	\Ingredient{1 teaspoon baking soda}
	\Ingredient{\fr12 teaspoon ginger}
	\Ingredient{\fr12 teaspoon nutmeg}
	\Ingredient{\fr14 teaspoon cinnamon}
	Whisk together.

	\Ingredient{\fr12 cup unsalted butter}
	\Ingredient{\fr34 cup brown sugar}
	\Ingredient{2 eggs}
	Beat together.

	\Ingredient{\fr73 cup (about 5) mashed bananas}
	\Ingredient{\fr12 cup chopped walnuts or pecans}
	Combine all the wet ingredients.
	Add the wet to the dry and stir to fully moisten.
	Pour batter into loaf pan.
	Bake \tooth{55}
\end{recipe}
