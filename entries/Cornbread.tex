\begin{recipe}{Corn Bread}{12 servings}{30 minutes prep, 20 minutes baking}

  Either:
  \begin{itemize}
    \item Start heating a cast iron skillet on the stove.
    \item Grease a muffin pan.
    \item Grease a 9\X9 inch pan.
  \end{itemize}
  Also preheat the oven to \tempf{425}.

  \Ingredient{\cornmeal{5}{4}}
  \Ingredient{\flour{whole wheat}{1/2}}
  \Ingredient{1 tablespoon sugar}
  \Ingredient{1 teaspoon baking powder}
  \Ingredient{\fr12 teaspoon baking soda}
  \Ingredient{\fr12 teaspoon salt}
  Mix together in a large bowl.

  \Ingredient{1 egg}
  \Ingredient{\milk{1}}
  \Ingredient{2 tablespoons buttermilk powder}
  Mix together.
  Add the wet ingredients to the dry and stir until just moistened.

  \Ingredient{2 tablespoons melted unsalted butter}
  Fold into the batter.
  Scrape the batter into the cooking pan and spread evenly.
  The baking time depends on the cooking pan; 12 minutes for muffins, 20 minutes in a skillet or glass pan.
  In any case, cook until it passes the toothpick test.

\end{recipe}
