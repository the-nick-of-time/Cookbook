\begin{recipe}{Savory Oatmeal with Sausage and Cheese}{1 serving}{10 minutes}

\Ingredient{\fr34 cup chicken broth}
\Ingredient{\fr12 cup rolled oats}
\Ingredient{2 precooked sausage links, sliced}
\Ingredient{pinch of ground cumin}
Bring to a boil in a small saucepan. Reduce heat to low and cook for 4-5 minutes, until oats thicken.

\Ingredient{1 tablespoon chopped green onion}
\Ingredient{2 tablespoons shredded Parmesan or Asiago cheese}
Stir in half of each, and put the other half on top.

\end{recipe}
