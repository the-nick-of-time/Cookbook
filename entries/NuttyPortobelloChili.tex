\begin{recipe}{Nutty Portobello Chili}{8 servings}{30 minutes prep, 40 minutes cooking}
% From Liz and Richenda Slaten at the gunbarrel chili cook-off

\Ingredient{\fr13 cup red wine}
Boil off the alcohol in a large pot.

\Ingredient{3 medium onions, chopped}
Add and cover pot to sweat them for about 5 minutes. Remove lid and saut\'e for about 5 minutes, until they are transparent.

\Ingredient{2 Portobello mushrooms, chopped (stems included)}
\Ingredient{4 garlic cloves, minced}
\Ingredient{2 tablespoons chili powder}
\Ingredient{1 tablespoon ground cumin}
Add and saut\'e for 2 more minutes.

\Ingredient{28 ounces crushed tomatoes}
\Ingredient{1 large sweet potato, peeled and cubed (\fr12 inch)}
\Ingredient{4 cups cooked or canned kidney or adzuki beans}
\Ingredient{\fr12 teaspoon black pepper}
\Ingredient{4 cups water}
Add, mix well, and cover. Simmer until the sweet potato is tender, about 30 minutes.

\Ingredient{1 bell pepper, finely chopped}
\Ingredient{1 cup chopped raw cashews}
\Ingredient{\fr12 cup fresh cilantro, chopped}
Remove from heat and add. Adjust amount of water and pepper.

\end{recipe}
