\begin{recipe}[AppleDumplings]{Apple Dumplings}{4 servings}{45 minutes}
% From the recipe book
\Ingredient{\fr23 cup sugar}
\Ingredient{\fr32 cups water}
\Ingredient{2 tablespoons butter}
\Ingredient{\fr14 teaspoon cinnamon}
Boil for 3 minutes.

\Ingredient{4 apples}
Core.

\Ingredient{\fr13 cup brown sugar}
\Ingredient{1 teaspoon cinnamon}
\Ingredient{2 teaspoons butter}
Fill the apple cavity, with the butter dotting the tops. Place in a deep baking dish.
Pour the syrup over and bake at 425\0F until apples are soft, around 20 minutes. %actual time unknown, wasn't in the picture
\end{recipe}
