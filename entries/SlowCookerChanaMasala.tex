\begin{recipe}{Slow Cooker Chana Masala}{8 servings}{7 hours}
% From March 2016 edition of delicious living magazine

\Ingredient{1 large onion, finely diced}
\Ingredient{4 medium carrots}
\Ingredient{2 cloves minced garlic}
\Ingredient{1 inch piece of ginger, peeled and grated}
\Ingredient{1 tablespoon ground coriander}
\Ingredient{2 tablespoons ground cumin}
\Ingredient{1 teaspoon ground turmeric}
\Ingredient{1 teaspoon paprika}
\Ingredient{1 teaspoon garam masala}
\Ingredient{\fr12 teaspoon salt}
\Ingredient{\fr14 teaspoon cayenne pepper}
\Ingredient{28 ounces of canned crushed tomatoes}
\Ingredient{30 ounces canned chickpeas, drained and rinsed}
Add in a slow cooker and stir to combine. Cover and cook for 6-7 hours.

\Ingredient{2 cups brown basmati rice}
Prepare like usual (see \reciperef{Rice}).

\Ingredient{juice and zest of \fr12 lemon}
\Ingredient{Plain greek yogurt}
Stir the lemon juice and zest into the sauce just before serving. Serve atop the rice and with a dollop of yogurt.

\end{recipe}
