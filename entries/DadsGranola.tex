\begin{recipe}[DadsGranola]{Dad's Base Granola Recipe}{50 servings}{1 hour}
\preheat{350}
Line a large, rimmed baking sheet with parchment paper.

\Ingredient{8 cups rolled oats}
\Ingredient{\fr32 cup plain hulled sunflower seeds}
\Ingredient{\fr12 cup sliced almonds}
\Ingredient{\fr12 cup chopped walnuts}
\Ingredient{1 teaspoon cinnamon}
\Ingredient{\fr12 teaspoon nutmeg}
Combine in a large mixing bowl.

\Ingredient{1 cup coconut oil, melted}
\Ingredient{\fr12 cup honey}
\Ingredient{2 tablespoons brown sugar}
\Ingredient{\fr32 teaspoons vanilla}
Mix in a pouring container and drizzle over the oats.
Stir well to coat evenly.
Pour into the baking sheet and spread evenly.
Bake until golden, about 21-23 minutes, stirring halfway.
The granola will crisp up more as it cools.
Let it cool completely.

\Ingredient{\fr23 cup dried fruit}
Break up the chunks of granola and mix in the fruit.
Store in airtight containers.
\end{recipe}
