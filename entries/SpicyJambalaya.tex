\begin{recipe}{Spicy Jambalaya}{6 servings}{15 minutes prep, 8 hours cooking}
	% from december 2010 Sunflower Market magazine

\Ingredient{1 tablespoon olive oil}
\Ingredient{8 ounces chicken breast, cut into \fr32 inch pieces}
\Ingredient{1 large green bell pepper, cored and chopped}
\Ingredient{1 small red onion, chopped}
\Ingredient{2 celery stalks, diced}
\Ingredient{2 teaspoons minced garlic}
\Ingredient{28 ounces diced tomatoes with juice}
\Ingredient{8 ounces tomato sauce}
\Ingredient{1 cup chicken broth}
\Ingredient{2 bay leaves}
\Ingredient{\fr12 teaspoon dried thyme}
\Ingredient{1 teaspoon dried oregano}
\Ingredient{\fr12 teaspoon cayenne pepper}
\Ingredient{\fr34 cup uncooked brown rice}
Mix well in a 4+ quart slow cooker. Cover and cook on low for 8 hours or high for 4 hours.

\Ingredient{1 pound shelled medium shrimp}
\Ingredient{2 tablespoons finely chopped parsley}
Add and cook an additional 10 minutes. Season with salt and pepper.

\end{recipe}
