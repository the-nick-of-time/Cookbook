\begin{recipe}[BeanDip]{Black Bean Dip}{A decent dip bowl}{15 minutes}

	\Ingredient{1 16oz can black beans, drained and rinsed}
	\Ingredient{\fr12 cup prepared salsa, hot or mild}
	\Ingredient{2 tablespoons lime juice (about \fr12 lime)}
	\Ingredient{\fr13 cup chopped fresh cilantro (about \fr13 bunch; make with Corn and Bean Salad on page \pageref{CornSalad} to use the rest)}
	\Ingredient{\fr32 teaspoon ground cumin}
	\Ingredient{salt and pepper to taste, about \fr14 teaspoon}
	Blend together and you have it. Most of the time spent is in washing and isolating the leaves of the cilantro.

\end{recipe}
