\begin{recipe}[BranMuffins]{Bran Muffins}{18 muffins}{1.5 hour}
\Ingredient{2 ounces unsalted butter, melted}
\Ingredient{\fr12 cup dark brown sugar}
\Ingredient{\fr14 cup molasses or honey}
Mix in large bowl.
\Ingredient{2 eggs}
\Ingredient{\milk{1}}
Beat together. Add to sugar.
\Ingredient{\fr32 cup bran cereal (All-Bran)}
\Ingredient{1 cup raisins}
Add and let stand.
\Ingredient{4 ounces nuts, chopped}
Add.
\Ingredient{\flour{\ww}{1/2}}
\Ingredient{\flour{\ap}{1/2}}
\Ingredient{\fr34 teaspoons salt}
\Ingredient{\fr32 teaspoons baking soda}
Sift together. Add to the wet ingredients all at once.
Spoon into tins, \fr23 full.
Bake at 400\0F until springy, about 15 minutes. Cool.
\end{recipe}
