\begin{recipe}[SpaghettiSauce]{Spaghetti Sauce}{8 person $\times$ meals}{60 minutes}

	\Ingredient{2 Italian pork sausages} %Half a pound?
	\Ingredient{1 pound ground beef}
	Brown the meats on medium heat.

	\Ingredient{1 large onion}
	Add onion, cook until it is translucent.

	\Ingredient{Several cloves of garlic}
	\Ingredient{\fr14 teaspoon fennel seed}
	Add garlic and fennel and cook for a few more minutes.

	\newstep
	Divide the meat in half, setting aside the other half for later batches.

	\Ingredient{4 cans diced tomato}
	\Ingredient{1 can tomato paste}
	\Ingredient{1 tablespoon oregano}
	\Ingredient{2 teaspoons basil}
	\Ingredient{1 teaspoon thyme}
	\Ingredient{2 Bay leaves}
	\Ingredient{\fr12 teaspoon marjoram (optional, can replace with more oregano)}
	\Ingredient{\fr14 teaspoon ground mustard}
	\Ingredient{\fr12 teaspoon dill}
	Add the tomato and the herbs. Simmer for a while until you get the desired consistency.

\end{recipe}
