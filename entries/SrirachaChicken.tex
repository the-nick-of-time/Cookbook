\begin{recipe}{Sriracha Chicken Bites}{3 servings}{90 minutes prep, 25 minutes cooking}

	\Ingredient{\fr32 pounds boneless chicken breasts or thighs in 3-cm chunks}
	\Ingredient{\fr32 cup buttermilk}
	\Ingredient{4 cloves minced garlic} % doubled from recommended 2
	\Ingredient{Salt, a large pinch}
	Marinate the chicken in the buttermilk mixture for 30--60 minutes.

	\Ingredient{3 tablespoons flour}
	\Ingredient{\fr12 teaspoon smoked paprika}
	\Ingredient{\fr12 teaspoon salt}
	\Ingredient{\fr14 teaspoon black pepper}
	Combine the dry ingredients into a coating mixture. Drain the chicken, then cover it thoroughly in the coating.

	\Ingredient{4 cups crushed potato chips/corn flakes}
	Dredge the chicken in the crunchy stuff.

	Bake for 18--24 minutes, until the chicken is fully cooked.

	\Ingredient{\fr14 cup Sriracha}
	\Ingredient{1 tablespoon rice vinegar}
	\Ingredient{\fr13 cup honey}
	\Ingredient{\fr12 teaspoon garlic powder} % doubled from recommended 1/4
	Combine and heat the ingredients in a small saucepan for 6--8 minutes. When the chicken is done, mix them together and serve.

\end{recipe}
