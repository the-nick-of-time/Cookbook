\begin{recipe}[CreamBeans]{Green Beans with Cream}{4 servings}{15 minutes}
% https://cooking.nytimes.com/recipes/5384-haricots-vert-a-la-creme-green-beans-in-cream
\Ingredient{1 pound green beans}
Steam lightly. Go easy here, since they're being cooked again soon.
% The original recipe says to boil them but I disagree, that drains flavor

\Ingredient{2 tablespoons butter}
\Ingredient{\cream{1/2}}
\Ingredient{\fr18 teaspoon grated nutmeg}
\Ingredient{\fr12 teaspoon dried tarragon}
\Ingredient{Salt and pepper to taste}
Drain the beans and put in a saucepan.
Add butter, cream and spices.
Bring to a boil, stirring frequently.
Cook briefly until the cream is slightly reduced and thickened.
\end{recipe}
