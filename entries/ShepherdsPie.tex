\begin{recipe}[ShepherdsPie]{Shepherd's Pie}{6 servings}{1 hour}
% Apparently this was made by Grandma Thurmes the night Alan Shepard went to space.
Preheat oven to \tempf{375}.

\Ingredient{1 pound beef, ground or diced}
\Ingredient{1 onion, diced}
\Ingredient{\fr14 teaspoon pepper}
Brown the meat and half the onion in a cast-iron pot.
Season with salt and pepper.
Remove meat to a plate.

\Ingredient{2 stalk celery}
\Ingredient{1 cup carrot, diced}
Saut\'e in the fat with the other half of the onion until tender.

\Ingredient{4 clove garlic, chopped}
\Ingredient{\fr14 cup flour}
Add and cook for 1 minute.

\Ingredient{2 cup beef broth}
\Ingredient{2 tablespoon tomato paste}
\Ingredient{\fr14 teaspoon thyme}
\Ingredient{\fr14 teaspoon rosemary}
\Ingredient{1 bay leaf}
Add and simmer for 5 minutes.

\Ingredient{\fr12 cup frozen peas}
\Ingredient{\fr12 cup frozen corn}
Remove bay leaf and stir in.

%Make biscuit dough as below:

\Ingredient{\flour{\ap}{1}}
\Ingredient{\fr12 teaspoon salt}
\Ingredient{2 teaspoon baking powder}
Mix together.

\Ingredient{2 tablespoon butter, chilled}
Cut into the flour.

\Ingredient{\milk{3/8}}
Mix in with fork until fully moistened.

\Ingredient{shredded cheese (optional)}
Pat out the dough on a floured surface to fit the pot.
Place on top of the stew, with cheese sprinkled on top if desired.
Put lid on, put in oven, and bake for 25 minutes.
Remove lid and bake for 5 more minutes to brown the top.

\end{recipe}
