\begin{recipe}{Green Tea Smoothie}{2 servings}{15 minutes}
	% From may 2011 Sunflower Markets magazine

\Ingredient{\fr12 cup boiling water}
\Ingredient{2 green tea bags}
Steep tea for 3 minutes, then let tea cool for 2-3 more minutes.

\Ingredient{2 small bananas, frozen or fresh}
\Ingredient{1 cup frozen strawberries}
\Ingredient{1 cup yogurt or milk}
Combine in blender. Pour in tea and pur\'ee until smooth.

\Ingredient{2 teaspoons agave nectar}
Add sweetener and any add-ins like protein powder, then blend again.

\end{recipe}
