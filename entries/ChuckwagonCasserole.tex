\begin{recipe}{Chuckwagon Casserole}{6 servings}{20 minutes prep, 40 minutes cooking}

%\preheat[beginning]{350} if using oven, else prepare crock pot.

\Ingredient{1 - 1.5 pounds ground beef}
\Ingredient{1 onion}
\Ingredient{1 teaspoon thyme}
Brown beef and onion in oven-safe pan. Cook off the liquid.

\Ingredient{about 4 pounds potatoes, sliced thin}
\Ingredient{3--4 ounces sliced or shredded cheese}
\Ingredient{salt and pepper}
Alternate layers of cheese and potato, with a sprinkling of salt and pepper on each potato layer.

\Ingredient{1 can baked beans}
Pour and spread baked beans over the top. Cover and put the pan in the oven for 30 minutes at \tempf{350} or until the potatoes are cooked.

\end{recipe}
