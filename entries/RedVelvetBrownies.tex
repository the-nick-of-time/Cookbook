\begin{recipe}{``Red" Velvet Brownies}{16 servings}{20 minutes prep, 40 minutes cooking}

Grease a 9\X9 inch pan and preheat the oven to 350\0F.

\Ingredient{\fr12 cup butter}
\Ingredient{\fr32 cup granulated sugar}
Beat together until fluffy.

\Ingredient{2 eggs, room temperature}
\Ingredient{4 teaspoons vanilla extract}
\Ingredient{3 tablespoons unsweetened cocoa powder}
Add and stir until completely blended.

\Ingredient{\flour{\ap}{5}{4}}
Mix in flour a bit at a time, until just blended. Don't overmix as the brownies turn out too tough.

\newstep
Pour the mix into the pan and bake \tooth{30-40}. Once they're done, let them cool for 20 minutes unless you just can't wait.

\end{recipe}
