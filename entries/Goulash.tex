\begin{recipe}{Goulash}{8 servings}{60 minutes}

\Ingredient{\fr12 pound bacon}
In a large pot ($\ge$ 5 quarts),
cook bacon until the pot is greased.

\Ingredient{1 onion}
Dice an onion.
Add it, cook until translucent.

\Ingredient{1 pound ground beef}
Add beef and cook it through.

\Ingredient{4 cans (56 ounces) diced tomato}
\Ingredient{1 can (6 ounces) tomato paste}
\Ingredient{1 pound frozen peas \& carrots or other vegetables}
\Ingredient{\fr12 teaspoon thyme}
\Ingredient{\fr12 teaspoon sage}
\Ingredient{\fr12 teaspoon basil}
\Ingredient{\fr14 teaspoon tarragon}
\Ingredient{\fr14 teaspoon paprika}
\Ingredient{Some garlic}
Add the vegetables and spices. Stir thoroughly.

\Ingredient{1 pound pasta of some type}
Add the pasta. Cook, stirring occasionally, until the pasta is done, about 30 minutes.

\end{recipe}
