\begin{recipe}{Pumpkin Cookies}{45 cookies}{40 minutes prep, 10 minutes baking}

Preheat oven to \tempf{375}. Prepare some cookie sheets with parchment paper.

\Ingredient{\flour{all-purpose}{5/2}}
\Ingredient{1 teaspoon salt}
\Ingredient{1 teaspoon baking soda}
\Ingredient{1 teaspoon cinnamon}
\Ingredient{\fr12 teaspoon nutmeg}
\Ingredient{\fr12 teaspoon ground cloves}
\Ingredient{\fr18 teaspoon ground ginger}
In a small bowl, sift together dry ingredients.

\Ingredient{\fr12 cup butter}
\Ingredient{1 cup brown sugar}
\Ingredient{\fr12 cup granulated sugar}
In a large bowl, beat together butter and sugars.

\Ingredient{15 ounces (1 can) pumpkin}
\Ingredient{1 teaspoon vanilla extract}
To the wet ingredients, add the pumpkin and vanilla and beat until homogenous.

\Ingredient{1 egg}
Add the egg and beat until homogenous.

\newstep
Add the dry ingredients in parts to the wet ingredients. Mix fully.

\Ingredient{\fr32 cups dark chocolate chips or chunks}
Stir in chocolate chips.

\newstep
Place spoonfuls of the mix onto the cookie sheets. Bake for 11 minutes, until done through.


\end{recipe}
