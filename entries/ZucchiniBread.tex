\begin{recipe}[ZucchiniBread]{Zucchini Bread}{2 loaves}{1 hour}
% Sue Rothchild's recipe
Preheat the oven to 350\0F. Prepare two loaf pans.

\Ingredient{2 eggs}
\Ingredient{\fr34 cup canola oil}
\Ingredient{\sugar{2/3}}
\Ingredient{1 teaspoon vanilla}
Mix together in a large mixing bowl.

\Ingredient{2 cup zucchini, grated}
\Ingredient{\fr12 cup chopped nuts}
Stir in to the wet ingredients.

\Ingredient{\flour{\ap}{2}}
\Ingredient{\flour{\ww}{1}}
\Ingredient{\fr12 teaspoon salt}
\Ingredient{1 teaspoon baking soda}
\Ingredient{\fr12 teaspoon baking powder}
\Ingredient{2 teaspoon cinnamon}
\Ingredient{\fr12 teaspoon nutmeg}
\Ingredient{\fr12 teaspoon cloves}
Mix together in a separate bowl.

Add the dry ingredients to the wet by parts. Pour into the loaf pans, about \fr23 full to allow for rising.

\end{recipe}
