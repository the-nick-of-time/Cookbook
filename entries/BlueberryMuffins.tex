\begin{recipe}[BlueberryMuffins]{Blueberry Muffins}{3 dozen}{1 hour}
    Preheat the oven to 400\0F and grease some muffin tins.

    \Ingredient{\flour{\ww}{4}}
    \Ingredient{1 teaspoon salt}
    \Ingredient{\fr13 cup sugar}
    \Ingredient{1 tablespoon brewer's yeast}
    Combine.
    
    \Ingredient{1 cup milk}
    Stir in enough milk to make a stiff dough.

    \Ingredient{\fr14 cup soybean oil}
    \Ingredient{2 eggs, lightly beaten}
    \Ingredient{1 cup blueberries}
    Stir in. Spoon into muffin tins, about \fr23 full. Bake \tooth{30}.

\end{recipe}
