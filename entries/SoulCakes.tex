\begin{recipe}[SoulCakes]{Soul Cakes}{12 servings}{1 hour}
% https://old.reddit.com/r/WitchesVsPatriarchy/comments/17j59wn/soul_cakes/k6yvvjm/
\preheat{full}{360}

\Ingredient{100 grams (7 tablespoons) butter, softened}
\Ingredient{100 grams caster sugar}
Cream together until light and fluffy.

\Ingredient{2 egg yolks}
Whisk in.

\Ingredient{\flour{\ap}{3/2}}
\Ingredient{pinch turmeric}
\Ingredient{1 teaspoon allspice}
\Ingredient{\fr12 teaspoon ginger}
Add to butter mixture.

\Ingredient{\milk{1/6}}
Add enough to make a dough that holds together.

\Ingredient{50 grams raisins}
Fold in.

Turn out onto a floured surface.
Roll to 1 centimeter thick.
Cut circles and place on parchment paper on baking pan.
Score in a cross pattern.

Bake for 25 minutes or until golden, firm, and slightly puffed.


\end{recipe}
