\begin{recipe}[BaseDessertBread]{Base Dessert Bread}{12 servings}{25 minutes prep + 1 hour rising}
	% https://www.kingarthurbaking.com/recipes/cinna-buns-recipe
	\Ingredient{\milk{1}}
	\Ingredient{2 eggs, at room temperature}
	\Ingredient{5 tablespoons (74 g) unsalted butter, softened}
	\Ingredient{\flour{all-purpose}{7/2}}
	\Ingredient{\fr74 teaspoon (11 g) salt}
	\Ingredient{\fr12 cup (99 g) granulated sugar}
	\Ingredient{\yeast{5}{2}}
	Mix and knead together to make a smooth, soft dough.
	Place the dough in a lightly oiled bowl and let rise until doubled in volume.

	After this point what you do with it is determined by what you're using it for.
\end{recipe}
