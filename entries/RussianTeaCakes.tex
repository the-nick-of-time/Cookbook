\begin{recipe}[RussianTeaCakes]{Russian Tea Cakes}{2 dozen}{20 minutes prep, 10 minutes baking}
\Ingredient{1 cup butter}
\Ingredient{\fr12 cup powdered sugar}
\Ingredient{1 teaspoon vanilla extract}
Mix thorougly.

\Ingredient{\flour{all-purpose}{9/4}}
\Ingredient{\fr14 teaspoon salt}
Mix in by parts.

\Ingredient{\fr34 cup chopped nuts}
Add.
Roll into 1" balls.
Bake at \tempf{400} until set, about 10 minutes.
They won't brown or spread at all.
Roll in powdered sugar while warm, then again shortly before serving.

\end{recipe}
