\begin{recipe}{Washtunkala Cast-iron Kettle Stew}{6 servings}{40 minutes}
% From march/april Encompass magazine

\Ingredient{\fr32 pounds buffalo cut into \fr32 inch cubes}
\Ingredient{3 tablespoons olive oil}
Saut\'e the meat in oil in a large pan over high heat, until browned.

\Ingredient{Salt and pepper to taste}
\Ingredient{\fr14 cup fresh thyme}
\Ingredient{12 ounces of pearl onions}
\Ingredient{12 ounces of corn}
\Ingredient{\fr32 chopped mild green chilies}
Add the spices, then the vegetables. Saut\'e for 1 minute.

\Ingredient{4 cups buffalo stock}
\Ingredient{\fr12 cup roasted hulled sunflower seeds}
Add the stock and sunflower seeds. Reduce the heat and simmer for 8--10 minutes until the broth thickens up.


\end{recipe}
