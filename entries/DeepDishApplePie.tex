\begin{recipe}{Deep-Dish Apple Pie}{1 pie}{75+25 minutes prep, 55--65 minutes baking}

	Prepare a pie crust like that included in this document on \reciperef{PieCrust}. Beware that this can be time-consuming.

	\newstep
	\preheat{425}

	\Ingredient{7--8 cups sliced apples}
	\Ingredient{\sugar{1/2}}
	\Ingredient{2--3 tablespoons flour}
	\Ingredient{1 tablespoon lemon juice (optional)}
	\Ingredient{1 teaspoon cinnamon}
	\Ingredient{\fr12 teaspoon nutmeg}
	Combine and let stand for 15 minutes, stirring occasionally.

	\Ingredient{2 tablespoons butter, cut into small pieces}
	Pour the filling into the bottom crust and dot with the butter. Seal the pie with the top crust and cut vents in it.\\
	Bake for 15 minutes.
	Reduce the oven temperature to \tempf{350}, and bake until juices bubble through the vents, about 40 to 50 minutes more.

\end{recipe}
