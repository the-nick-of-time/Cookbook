\begin{recipe}[SquashCustard]{Squash Custard}{8 servings}{1 hour (ahead of time) prep, 40 minutes cooking}
	\Ingredient{1 winter squash of choice - kuri or buttercup/kabocha}
	Bake the squash skin-side down with some water until tender.
	Scrape out the flesh. This can be chilled for as long as necessary, at least a few days.

	\Ingredient{1 egg per cup of squash}
	\Ingredient{1 tablespoon butter per cup of squash}
	\Ingredient{1 tablespoon brown sugar per cup of squash}
	\Ingredient{\fr14--\fr34 cup milk per cup of squash}
	\Ingredient{\fr18 teaspoon salt}
	\Ingredient{\fr12 teaspoon cinnamon}
	\Ingredient{\fr18 teaspoon cloves}
	\Ingredient{\fr14 teaspoon allspice}
	\Ingredient{\fr14 teaspoon nutmeg}
	Whisk egg and preliminary amount of milk.
	Mix in with the squash, blending well.
	Bake at \tempf{350} until the inside doesn't jiggle, about 40 %????
	minutes.
\end{recipe}
