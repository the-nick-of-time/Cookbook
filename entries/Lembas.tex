\begin{recipe}{Lembas}{6 servings}{30 minutes prep, 12 minutes baking}
% http://theweepingtimelord.tumblr.com/post/39725517841/lembas-bread-lord-of-the-rings-authentic-elvish
%\begin{tengwar}
%	\Tlambe \TTacute \Tumbar \TTthreedots \Tsilme
%\end{tengwar}

%\tengwarquenya{\Tlambe \TTacute \Tumbar \TTthreedots \Tsilme}\\
Preheat oven to \tempf{425}.

\Ingredient{\flour{all-purpose}{5}{2} optionally substituting \flour{\ww}{1} if it's fine enough}
\Ingredient{1 tablespoon baking powder}
\Ingredient{\fr14 teaspoon salt}
Mix together in a large bowl.

\Ingredient{\fr12 cup butter}
Add butter and cut it in until the consistency is of small granules (easiest with food processor).

\Ingredient{\fr13 cup brown sugar}
\Ingredient{1 teaspoon cinnamon}
Add in the sugar and cinnamon and mix thoroughly.

\Ingredient{\fr12 teaspoon honey or maple syrup}
\Ingredient{\fr23 cup heavy whipping cream}
\Ingredient{\fr12 teaspoon vanilla extract}
\Ingredient{\fr14 teaspoon almond extract}
Add the liquids and stir (a fork works best) until a thick dough forms.

Roll out the dough to about 1 centimeter thick. Cut into 8-centimeter squares and transfer to a cookie sheet. Bake for about 12 minutes or until it is set and light golden.

Serve warm. If you keep them in the fridge, heat them in the toaster before eating.

% Let the \tengwarquenya{\Tlambe \TTacute \Tumbar \TTthreedots \Tsilme} cool completely before eating.
\end{recipe}
