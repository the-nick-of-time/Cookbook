\begin{recipe}[SloppyJoes]{Sloppy Joes}{6 servings}{30 minutes}
% marked as "Maggie's Recipe"
\Ingredient{1 pound ground beef}
\Ingredient{2 teaspoon onion powder}
\Ingredient{1 cup celery, diced}
Brown in a saucepan.

\Ingredient{1 cup ketchup}
\Ingredient{1 teaspoon lemon juice}
\Ingredient{2 teaspoon vinegar}
\Ingredient{\fr12 cup water}
\Ingredient{2 teaspoon brown sugar}
\Ingredient{\fr12 teaspoon dry mustard}
\Ingredient{\fr12 teaspoon salt}
Add, bring to a simmer, and simmer for 20 minutes.


\end{recipe}
