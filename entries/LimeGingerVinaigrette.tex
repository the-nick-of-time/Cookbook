\begin{recipe}{Lime-Ginger Vinaigrette}{8 servings}{5 minutes}
% from August 2014 delicious living magazine

\Ingredient{4 tablespoons chopped shallot}
\Ingredient{\fr32 tablespoon chopped fresh ginger}
\Ingredient{2 tablespoons minced serrano pepper}
\Ingredient{3 tablespoons chopped fresh cilantro}
\Ingredient{\fr12 cup fresh lime juice}
\Ingredient{\fr14 cup rice wine vinegar}
\Ingredient{3 tablespoons agave nectar}
\Ingredient{4 tablespoons soy sauce}
\Ingredient{3 tablespoons sweet chile sauce}
\Ingredient{2 teaspoons kosher salt}
\Ingredient{1 teaspoon black pepper}
Combine in a blender and pulse for 1 minute.

\Ingredient{\fr12 cup vegetable oil}
Slowly drizzle in oil and pulse to emulsify.

\end{recipe}
