\begin{recipe}[VegetarianBeanChili]{Vegetarian Bean Chili}{6 servings}{50 minutes}
%	https://www.cookincanuck.com/instant-pot-vegetarian-chili-recipe/

\Ingredient{2 teaspoons olive oil}
\Ingredient{1 yellow onion, chopped}
Saut\'e in the Instant Pot until softened.

\Ingredient{6 garlic cloves, minced}
\Ingredient{\fr14 cup chili powder}
\Ingredient{2 teaspoons dried oregano}
\Ingredient{2 teaspoons ground cumin}
Add and cook for 1 minute.

\Ingredient{1 tablespoon tomato paste}
Add and stir constantly for 1 minute.

\Ingredient{\fr74 cup vegetable broth}
\Ingredient{28 ounces crushed tomatoes}
\Ingredient{1 chipotle pepper}
\Ingredient{1 teaspoon adobo sauce from chipotle pepper jar}
\Ingredient{28 ounces black beans}
\Ingredient{28 ounces kidney beans}
\Ingredient{14 ounces great northern beans}
\Ingredient{\fr14 teaspoon salt}
Add and stir to combine.
Pressure cook on high for 10 minutes.
After the time has passed, wait 5 minutes then vent.

\Ingredient{Cilantro}
Stir in and serve.

\end{recipe}
