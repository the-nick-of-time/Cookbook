\begin{recipe}{Berry Crumble}{8 servings}{15 minutes prep, 60 minutes baking}

	\Ingredient{5-6 cups mixed berries (frozen)}
	\Ingredient{\fr14 cup quick-cooking tapioca}
	\Ingredient{\fr14 cup sugar}
	\Ingredient{1 teaspoon vanilla extract}
	\Ingredient{\fr12 teaspoon almond extract}
	\Ingredient{1 tablespoon lemon juice}
	Mix together in the cooking dish.

	\Ingredient{\fr32-2 cups rolled oats}
	\Ingredient{4 tablespoons butter}
	\Ingredient{\fr13 cup brown sugar}
	\Ingredient{\fr12 teaspoon cinnamon}
	Mix together in a separate dish, then top the berries.

	\newstep
	Bake for about an hour at 350\0F with the lid on.
	This will undoubtedly be less if you use unfrozen berries.
	When it starts boiling, take the lid off and let it bake for another 10-15 minutes before it's done.


\end{recipe}
