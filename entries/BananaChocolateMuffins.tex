\begin{recipe}[BananaChocolateMuffins]{Banana Chocolate Muffins}{12 muffins}{40 minutes}
	% https://www.reddit.com/r/religiousfruitcake/comments/pf6019/im_that_real_slave_type/hb4fpf8/
	\Ingredient{3 bananas}
	\Ingredient{1 egg}
	\Ingredient{vanilla extract}
	\Ingredient{\fr13 cup coconut oil}
	Mash together.
	\preheat{350}

	\Ingredient{1 tablespoon baking powder}
	\Ingredient{\fr32 cups flour}
	\Ingredient{\fr14 teaspoon salt}
	\Ingredient{\fr12 cup brown sugar}
	Mix in a large bowl.
	Add the wet ingredients to the dry and stir until fully moistened.

	\Ingredient{\fr12 cup chocolate chips}
	\Ingredient{\fr14 cup chopped almonds}
	Stir into the batter.
	Pour into your muffin pan and bake \tooth{10}
\end{recipe}
