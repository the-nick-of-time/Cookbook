\begin{recipe}[Apfelkuchen]{Apfelkuchen}{12 servings}{10 minutes prep, 30 minutes rising, 25 minutes baking}
	\Ingredient{\flour{\ap}{9}{4}}
	\Ingredient{\fr38 cup granulated sugar}
	\Ingredient{\yeast{5}{2}}
	\Ingredient{pinch salt}
	Sift together.

	\Ingredient{\fr98 cup lukewarm milk}
	\Ingredient{1 teaspoon vanilla}
	\Ingredient{6 tablespoons butter, softened}
	Put wet ingredients together.
	Vigorously mix with the rest, even a blender works.
	Let dough rest for 5 minutes.

	\Ingredient{5 tart apples}
	Slice the apples.
	Grease a 10" round pan.
	Pat the dough into the pan.
	Place apples on the dough and push in lightly.
	Let rise in a warm place for 30 minutes.

	\Ingredient{\flour{\ap}{3}{4}}
	\Ingredient{\fr12 cup brown sugar}
	\Ingredient{\fr12 cup granulated sugar}
	\Ingredient{pinch salt}
	\Ingredient{1 teaspoon cinnamon}
	\Ingredient{\fr12 teaspoon nutmeg}
	\Ingredient{\fr12 cup butter, refrigerator temperature}
	Combine all dry ingredients, then cut in butter until mixture is crumbly.
	Sprinkle over the top of the bread.
	Heat oven to 400\0F.
	Bake for 20-30 minutes, or until brown.
\end{recipe}
