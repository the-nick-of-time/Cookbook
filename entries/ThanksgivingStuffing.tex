\begin{recipe}[Stuffing]{Thanksgiving Stuffing}{10 servings}{time}

	\Ingredient{16 cups dried bread chunks}
	\Ingredient{3 cups poultry stock}
	\Ingredient{2 diced onions}
	\Ingredient{5 stalks diced celery}
	\Ingredient{\fr34 cup dried cranberries}
	\Ingredient{\fr34 cup pine nuts}
	\Ingredient{3 tablespoons sage-fresh and/or dried}
	\Ingredient{2 teaspoons pepper}
	\Ingredient{\fr32 tablespoon salt}
	Break bread into chunks and put in a large bowl. Dice onion and celery and add them. Rub the sage in, or otherwise reduce it to smallish bits.

	\Ingredient{2 tablespoons melted butter}
	Mix the butter with any stuffing that remains after filling the bird. Bake it later, with the squash custard.

\end{recipe}
