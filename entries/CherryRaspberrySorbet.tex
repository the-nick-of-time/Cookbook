\begin{recipe}{Cherry Raspberry Sorbet}{2-3 servings}{20 minutes}
	% from june 2012 Sunflower Market magazine

\Ingredient{\fr34 cup apple juice}
\Ingredient{\fr14 cup frozen cherries}
\Ingredient{\fr14 cup frozen raspberries}
\Ingredient{\fr14 teaspoon almond extract}
Blend together until smooth.

\Ingredient{1 large frozen banana}
Add chunks and blend until mixture is thick.

\end{recipe}
