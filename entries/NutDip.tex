\begin{recipe}{Nut Dip}{serves many}{30 minutes prep, 2 hours chilling}
% from march 2018 delicious living magazine

\Ingredient{\fr34 cup hazelnuts}
Combine with boiling water and let sit until cool, then drain and dry.

\Ingredient{\fr12 cup blanched almonds}
Blend with the hazelnuts until finely ground.

\Ingredient{4 tablespoons olive oil}
\Ingredient{2 tablespoons sherry vinegar}
\Ingredient{1 teaspoon hot paprika}
\Ingredient{1 teaspoon agave nectar}
\Ingredient{1 teaspoon salt}
\Ingredient{\fr14 teaspoon cayenne pepper}
\Ingredient{\fr14 cup parsley}
\Ingredient{4 cloves garlic}
Add to blender, and blend until smooth. Cool for at least 2 hours. Serve with pita or veggies.

\end{recipe}
