\begin{recipe}{Scones}{8 servings}{20 minutes prep, 20 minutes baking}

\Ingredient{\flour{whole wheat}{2}}
\Ingredient{\fr52 teaspoon baking powder}
\Ingredient{\fr14 cup granulated sugar}
\Ingredient{\fr12 teaspoon salt}
Mix together dry ingredients.

\Ingredient{\fr43 cup (306 g) heavy cream}
Stir in cream. Mix until the dough starts to come together. The dough will be sticky. Put the dough on a floured surface and knead briefly, until it is completely together. Form into an 8-inch circle.

\Ingredient{1 tablespoon melted butter}
\Ingredient{\fr32 tablespoons sugar}
\Ingredient{sprinkle nutmeg}
Brush the disk with the butter and sprinkle the sugar over top. Cut into wedges and place onto a baking sheet with parchment paper. Bake for 17 minutes at \tempf{425} or until golden brown.

\end{recipe}
