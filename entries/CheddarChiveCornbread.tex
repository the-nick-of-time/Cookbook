\begin{recipe}{Cheddar Chive Cornbread}{8 servings}{40 minutes}
% from march 2018 delicious living magazine

\Ingredient{4 tablespoons unsalted butter}
Melt butter in a 9-inch cast iron skillet while oven preheats to \tempf{400}. Remove 2 tablespoons of melted butter from pan and set aside. Place pan back in oven.

\Ingredient{\fr32 cups cornmeal}
\Ingredient{\flour{\ap}{1}{2}}
\Ingredient{1 tablespoon ground flaxseed}
\Ingredient{1 tablespoon baking powder}
\Ingredient{\fr12 teaspoon salt}
Stir together in a medium bowl.

\Ingredient{\milk{1}}
\Ingredient{2 tablespoons honey}
\Ingredient{2 eggs, lightly beaten}
\Ingredient{1 cup shredded cheddar cheese}
\Ingredient{\fr34 cup finely chopped chives}
Stir into the dry ingredients along with the reserved butter.

Remove skillet from oven, and swirl butter to coat. Pour in batter, and distribute evenly. Bake for 20--22 minutes or until edges are golden brown and cornbread is set.

\end{recipe}
