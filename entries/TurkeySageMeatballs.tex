\begin{recipe}{Turkey-Sage Mini Meatballs with Smoky Honey Glaze}{5 servings}{30 minutes}
	% From october 2015 delicious living magazine

\preheat{425}

\Ingredient{1 egg yolk}
\Ingredient{\fr12 teaspoon honey}
\Ingredient{\fr12 cup finely chopped onion}
\Ingredient{\fr13 cup flaxseed meal}
\Ingredient{\fr34 teaspoon dried sage}
\Ingredient{\fr12 teaspoon dried thyme}
\Ingredient{\fr12 teaspoon smoked paprika}
\Ingredient{\fr12 teaspoon salt}
\Ingredient{\fr14 teaspoon black pepper}
\Ingredient{\fr54 pound ground turkey}
Mix spices, then add turkey and mix well. Shape into 1-inch balls (about \fr34 ounce each), using wet hands as needed to prevent sticking.

\Ingredient{\fr12 cup finely chopped walnuts}
\Ingredient{\fr14 cup flaxseed meal}
Combine in a shallow dish. Roll meatballs to coat. Place meatballs on a rimmed baking sheet.
Start baking meatballs for 5 minutes.

\Ingredient{\fr12 cup honey}
\Ingredient{\fr12 teaspoon smoked paprika}
\Ingredient{2 tablespoons apple cider vinegar}
Combine in a small bowl. After the 5 minutes, remove meatballs from oven and brush the glaze evenly over them. Return to oven and bake 5--8 minutes more or until done. Serve with the juices from the baking pan and any remaining glaze.

\end{recipe}
