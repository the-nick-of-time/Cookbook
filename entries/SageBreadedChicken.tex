\begin{recipe}{Sage-Breaded Chicken, Spinach and Snap Pea Salad}{4 servings}{30 minutes}
% From march 2018 delicious living magazine

\Ingredient{\fr14 cup white balsamic vinegar}
\Ingredient{\fr12 cup olive oil}
\Ingredient{\fr12 cup flat-leaf parsley}
\Ingredient{\fr12 cup basil leaves}
\Ingredient{\fr14 cup grated Parmesan}
\Ingredient{\fr14 teaspoon salt}
Blend together until smooth. Makes \fr34 cup.

Preheat oven to \tempf{400}. Coat a 15\X10\X1-inch baking pan with nonstick spray.

\Ingredient{1 egg, lightly beaten}
\Ingredient{\fr14 cup sour cream}
Whisk together in a medium bowl.

\Ingredient{\fr34 cup bread crumbs}
\Ingredient{\fr32 teaspoon ground sage}
Stir together in a different medium bowl.

\Ingredient{1 pound chicken tenders}
Stir in with the egg mixture. Dredge in the breading and place on prepared pan. Bake for 12-15 minutes, turning after 5 minutes, or until chicken is done.

\Ingredient{5 ounces baby spinach}
\Ingredient{2 cups snap peas, chopped}
\Ingredient{2 cups strawberries, sliced}
\Ingredient{2 avocados, thinly sliced}
\Ingredient{\fr14 cup roasted unsalted almonds, chopped}
\Ingredient{\fr14 cup lightly packed fresh basil leaves, snipped}
Make a salad. Top with sliced chicken and drizzle with the vinaigrette.

\end{recipe}
