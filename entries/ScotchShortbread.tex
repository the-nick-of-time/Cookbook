\begin{recipe}[ScotchShortbread]{Scotch Shortbread}{20 servings}{40 minutes}
    Preheat oven to \tempf{325}.

    \Ingredient{1 cup butter}
    Cream in a medium bowl.

    \Ingredient{\flour{\ap}{2}}
    \Ingredient{\fr12 cup confectioners' sugar}
    \Ingredient{\fr14 teaspoon salt}
    Sift together.
    Blend into the butter.
    Pat the dough into an ungreased 9\X9 inch pan and press edges down.
    Pierce with a fork all the way through the dough every half-inch.
    Bake 25-30 minutes.
    Cut into squares while warm.
\end{recipe}
