\begin{recipe}[ZucchiniChocolateMuffins]{Zucchini Chocolate Muffins}{12 muffins}{60 minutes}
	% https://www.reddit.com/r/religiousfruitcake/comments/pf6019/im_that_real_slave_type/hb4jmli/
	\Ingredient{\fr32 cups flour}
	\Ingredient{\sugar{2/3}}
	\Ingredient{1 teaspoon baking soda}
	\Ingredient{1 teaspoon cinnamon}
	\Ingredient{\fr12 teaspoon salt}
	Combine in a large mixing bowl.
	\preheat{350}

	\Ingredient{1 egg}
	\Ingredient{\fr12 cup vegetable oil}
	\Ingredient{\milk{1/4}}
	\Ingredient{1 tablespoon lemon juice}
	\Ingredient{1 teaspoon vanilla extract}
	Mix in a separate bowl.
	Stir into the dry ingredients until moistened.

	\Ingredient{1 cup shredded zucchini}
	\Ingredient{\fr12 cup mini chocolate chips}
	Fold into the batter.
	Pour into pan, about \fr23 full.
	Bake \tooth{25}
\end{recipe}
