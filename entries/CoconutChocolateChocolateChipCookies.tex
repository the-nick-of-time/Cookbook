\begin{recipe}{Coconut Chocolate Chocolate Chip Cookies}{48 cookies}{40 minutes prep, 30 minutes baking}
	Preheat oven to \tempf{350}. Grease 2-3 cookie sheets.

	\Ingredient{2 ounces baking chocolate}
	\Ingredient{\fr14 cup coconut oil}
	Melt baking chocolate and coconut oil in the microwave.
	%If desired, you can mix in some coconut oil to help it melt.
	%If you do, reduce the amount of butter by an equal amount.


	\Ingredient{1 cup butter (2 sticks), softened}
	In a large bowl, beat the butter until creamy. It should be smooth, and have a little air trapped inside.

	\Ingredient{\fr23 cup granulated sugar}
	\Ingredient{\fr34 cup brown sugar}
	Add sugars to butter and beat together.

	\newstep
	Add melted chocolate to the mix.

	\Ingredient{2 large egg--beaten}
	\Ingredient{2 tablespoon milk or half \& half or water }
	\Ingredient{1 tablespoon vanilla extract}
	Add liquid ingredients to the mix and beat them together.

	\Ingredient{\flour{\ww}{3}{2}}
	\Ingredient{\flour{\ap}{1}}
	\Ingredient{\fr12 tsp baking soda}
	\Ingredient{1 tsp baking powder}
	\Ingredient{\fr14 tsp salt (\fr12 teaspoon if using unsalted butter)}
	Add dry ingredients to mix, \fr12 cup at a time.

	\Ingredient{1 cup nuts (walnuts or pecans)}
	\Ingredient{\fr32 cups rolled oats}
	Pulverize, with nut chopper or food processor. Add them to the mix.

	\Ingredient{1 cup chocolate chips}
	\Ingredient{1 cup dried fruit}
	\Ingredient{1 cup rolled oats}
	Stir into the mixture by hand. Form 1-inch slightly flattened balls and place them on the cookie sheets. Bake for 10-12 minutes.

\end{recipe}
