\begin{recipe}[Macaroons]{Macaroons}{10 servings}{5 minutes prep, 20 minutes cooking}
\Ingredient{\fr43 cup coconut}
\Ingredient{\fr13 cup sugar}
\Ingredient{2 tablespoons flour}
\Ingredient{\fr12 teaspoon almond extract}
\Ingredient{\fr18 teaspoon salt}
Mix together.

\Ingredient{2 egg whites}
Add and beat in.
Bake 20 minutes at 325\0F, until lightly browned.
\end{recipe}
