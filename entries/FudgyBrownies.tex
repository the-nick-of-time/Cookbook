\begin{recipe}{Fudgy Brownies}{16 servings}{20 minutes prep, 20 minutes baking}
% https://www.reddit.com/r/food/comments/8ksr4i/homemade_fudgy_chocolate_brownies/dzao4s2/

Preheat oven to \tempf{325}. Line a 8\X8 inch pan with parchment paper.

\Ingredient{250 grams sugar}
\Ingredient{140 grams butter}
\Ingredient{80 grams cocoa powder}
Heat in a medium bowl in the microwave until butter melts. Mix well.

\Ingredient{10 milliliters vanilla extract}
\Ingredient{2 eggs}
Add one at a time. Mix thoroughly after each.

\Ingredient{\flour{all-purpose}{3}{8}}
Add, mix thoroughly, then beat another 40 strokes.

\Ingredient{200 grams chocolate chips}
Fold in chocolate chips. Spread into the prepared pan. Bake \tooth{20}

\end{recipe}
