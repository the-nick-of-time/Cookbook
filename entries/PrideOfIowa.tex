\begin{recipe}{Pride of Iowa Cookies}{72 cookies}{35 minutes prep, 15 minutes baking}

	\Ingredient{\brownsugar{1}}
	\Ingredient{\sugar{1}}
	\Ingredient{1 cup butter}
	Cream together.

	\Ingredient{2 eggs}
	\Ingredient{1 teaspoon vanilla}
	Beat in eggs and vanilla.

	\Ingredient{1 teaspoon baking soda}
	\Ingredient{1 teaspoon baking powder}
	\Ingredient{1 teaspoon salt}
	\Ingredient{\flour{all-purpose}{2}}
	Mix together dry ingredients, and add them to the wet ingredients. Mix together thoroughly.

	\Ingredient{2 cup coconut}
	\Ingredient{\fr12 cup chopped nuts (black walnuts are best, and I use a generous half cup)}
	\Ingredient{2 cup quick-cooking oatmeal (optional, not included in refrigerator variant)}
	Add the nuts, oatmeal, and coconut. Form the dough into two rolls, wrap in plastic, and chill until the dough is firm.\\
	When ready, slice the dough into cookies about 5 millimeters thick and bake at \tempf{375}.
	At 18 cookies to a sheet, bake for 10 minutes.
	If the cookies are thicker or you have more on a single sheet, they take longer to bake.
	If they're sparse and thin, check after 6 minutes and finish with 1--4 extra minutes.

\end{recipe}
