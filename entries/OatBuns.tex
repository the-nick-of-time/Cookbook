\begin{recipe}[OatBuns]{Oat Buns}{16 rolls}{15 minutes prep, 25 minutes baking}
% https://www.kingarthurbaking.com/recipes/wheat-oat-flax-buns-recipe
\Ingredient{\flour{all-purpose}{3}{2}}
\Ingredient{\flour{whole wheat}{3}{4}}
\Ingredient{\fr12 cup (45 g) rolled oats}
\Ingredient{\fr12 cup (43 g) whole flax meal}
\Ingredient{\yeast{5}{2}}
\Ingredient{\fr32 teaspoon (9 g) salt}
\Ingredient{4 tablespoons (57 g) butter, melted}
\Ingredient{1 large egg yolk, white reserved for wash}
\Ingredient{3 tablespoons (64 g) honey}
\Ingredient{\fr14 cup (57 g) orange juice\footnotemark}
\Ingredient{\fr78 cup (198 g) lukewarm water}

Combine in a large bowl, mixing until a soft dough forms.
Knead until smooth and satiny.

Place in a greased bowl and let rise until about doubled.
Deflate the dough and divide into 16 pieces (about 50 g each).
Place on a baking sheet, cover in plastic, and let rise for about 90 minutes, until puffy-looking.
Preheat the oven to \tempf{350}.

\Ingredient{1 egg white (from above)}
\Ingredient{2 tablespoons cold water}
\Ingredient{rolled oats or flax seeds}
Whisk the egg and water together to make an egg wash.
Uncover the buns and brush the tops with egg wash.
Sprinkle the oats or seeds on top and bake for 21 to 25 minutes, until golden brown.
Remove from the oven and transfer to a rack to cool.
\end{recipe}
\footnotetext{Doesn't add its own flavor, but cuts the bitterness of the whole wheat flour}
