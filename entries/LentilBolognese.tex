\begin{recipe}{Lentil Bolognese}{6 servings sauce}{15 minutes prep, 30 minutes cooking}

\Ingredient{2 tablespoons olive oil}
\Ingredient{1 onion, finely diced}
\Ingredient{1 stalk celery, finely diced}
\Ingredient{3 large carrots, peeled and chopped}
Heat oil in a large skillet, then add the vegetables. Cook until softened, 5--10 minutes.

\Ingredient{7 cloves garlic, minced}
Add garlic and cook for 30 seconds.

\Ingredient{6 ounces (1 can) tomato paste}
\Ingredient{15 ounces (1 can) tomato sauce}
\Ingredient{30 ounces (2 cans) diced tomatoes}
\Ingredient{2 tablespoons dried basil}
\Ingredient{1 teaspoon dried oregano}
\Ingredient{\fr14 teaspoon baking soda}
\Ingredient{Salt and pepper to taste}
Add tomato paste and stir it in for a minute. Add tomato sauce, diced tomatoes with juice, and seasonings.

\Ingredient{2 cups red lentils, rinsed and picked over}
\Ingredient{2 cups water}
Add lentils and water. Simmer until lentils are softened, 20--30 minutes.

\end{recipe}
