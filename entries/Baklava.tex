\begin{recipe}[Baklava]{Baklava}{1 large pan}{2 hours}
% https://old.reddit.com/r/food/comments/a94w8a/homemade_baklava/ecgjzwk/
\Ingredient{2 eggs}
\Ingredient{200 mL milk}
\Ingredient{100 mL yogurt}
\Ingredient{200 mL vegetable oil}
\Ingredient{1 tablespoon vinegar}
\Ingredient{Pinch of salt}
% Originally called for 1/2 packet which is not a thing here
\Ingredient{\fr52 teaspoon of baking powder}
Mix together.

\Ingredient{\flour{all-purpose}{20/3}}
Mix in the flour by fractions, mixing thoroughly each time.
When the dough is all combined, cut it into 45 pieces.

\Ingredient{Cornstarch}
\Ingredient{Crushed walnuts, almonds, or hazelnuts}
Preheat your oven to \tempf{325}.
Roll each piece in starch to the size of your cooking tray.
Place on the tray and sprinkle some crushed nuts on it.

\Ingredient{250 g butter}
When all the dough has been layered, melt the butter and spread it evenly across the top.

\Ingredient{1 L water}
\Ingredient{800 g white sugar}
Boil for 20 minutes.

\Ingredient{4--5 drops of lemon juice}
Add the lemon, then boil another 5 minutes. Let cool for another 5 minutes, then add to the baked dough.

\end{recipe}
