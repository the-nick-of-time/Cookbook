\begin{recipe}{Beth's Butter \& Sugar Cookies}{24 cookies}{25 minutes prep, 2 hours chilling, 30-90 minutes rolling/baking}

	\Ingredient{1 cup soft butter}
	\Ingredient{\fr32 cups powdered sugar}
	Cream together butter and sugar.

	\Ingredient{1 egg}
	\Ingredient{\fr32 teaspoon vanilla}
	Add egg and vanilla, and thoroughly cream together.

	\Ingredient{\flour{all-purpose}{3}}
	\Ingredient{\fr12 teaspoon baking soda}
	\Ingredient{\fr12 teaspoon cream of tartar}
	Sift the dry ingredients together, then slowly add to the wet ingredients.

	\newstep
	Cover and chill for at least 2 hours, until the dough is firm. Roll \fr14 inch thick between two pieces of plastic wrap to avoid adding more flour. Cut, decorate as desired, and lay on cookie sheets. Bake at 375\0F for
	% Our oven:
	3 minutes, then flip them and bake for another 3 minutes % go down to 3?
	or until fully baked but not browned.

\end{recipe}
