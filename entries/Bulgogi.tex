\begin{recipe}[Bulgogi]{Bulgogi}{10 servings}{Marinade plus 20 minutes}
	\Ingredient{3 tablespoons beef broth}
	\Ingredient{3 green onions, chopped}
	\Ingredient{4 cloves garlic, minced}
	\Ingredient{2 tablespoons sesame oil}
	\Ingredient{1 tablespoon sesame seeds, somewhat crushed}
	\Ingredient{\fr14 cup soy sauce}
	\Ingredient{\sugar{1/4}}
	\Ingredient{pinch of black pepper}
	\Ingredient{2 pounds finely shaved flank or sirloin (starting frozen makes this easier)}
	Mix together everything as a marinade for the meat. Let sit in the fridge for a while.

	Cook at medium-high heat on stove.
\end{recipe}
