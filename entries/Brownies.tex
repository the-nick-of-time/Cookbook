\begin{recipe}[Brownies]{Brownies}{16 brownies}{40 minutes}
\Ingredient{\flour{\ap}{3/4}}
\Ingredient{\fr14 teaspoon salt}
Sift together.

\Ingredient{\fr12 cup butter}
\Ingredient{1 cup sugar}
\Ingredient{2 eggs, beaten}
Cream together. Beat until smooth and fluffy.

\Ingredient{1 teaspoon vanilla}
\Ingredient{2 ounces baking chocolate, melted and cooled}
Add to the butter and stir.

Add flour by parts and mix well after each addition.
Bake in a greased 8" pan at 350\0F \tooth{20}.
\end{recipe}
