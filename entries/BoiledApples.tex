\begin{recipe}{Boiled Apples}{2 servings}{30 minutes}
  \Ingredient{2 apples, cored and sliced}
	\Ingredient{1 cup apple juice}
	\Ingredient{\fr14 cup brown sugar}
	\Ingredient{1 teaspoon cinnamon}
	\Ingredient{\fr12 teaspoon nutmeg}
	\Ingredient{\fr14 teaspoon ginger}
	Put in a small saucepan and heat to a simmer. Cook until apples are done.
\end{recipe}
