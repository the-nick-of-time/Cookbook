\begin{recipe}[DarkChocolateCheesecake]{Dark Chocolate Cheesecake}{12 servings}{2 hours}
\Ingredient{\fr32 cup vanilla wafers}
\Ingredient{6 tablespoons butter, melted}
\Ingredient{\fr14 cup sugar}
Crumble wafers, then mix well with the rest. Press into 9" round baking dish. Chill 5 to 10 minutes, or bake 10 minutes at 350\0F then chill.

\Ingredient{\fr32 pounds cream cheese}
\Ingredient{\fr34 cup sugar}
Beat together.
\Ingredient{3 eggs}
\Ingredient{1 teaspoon vanilla extract}
Beat into the cheese.
\Ingredient{5 ounces semisweet chocolate, melted in double boiler}
\Ingredient{1 cup sour cream}
Stir in and blend well.

Pour into crust. Bake 1 hour 15 minutes at 275\0F.

\end{recipe}
